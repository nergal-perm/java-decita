24  package ru.ewc.decita.input;
25  
26  import java.io.IOException;
27  import java.net.URI;
28  import java.nio.charset.StandardCharsets;
29  import java.nio.file.Files;
30  import java.nio.file.Path;
31  import java.nio.file.Paths;
32  import java.util.Collections;
33  import java.util.List;
34  import java.util.Map;
35  import java.util.stream.Collectors;
36  import java.util.stream.Stream;
37  import ru.ewc.decita.DecisionTable;
38  import ru.ewc.decita.Locator;
39  
40  /**
41   * I am the specific {@link ContentReader} that knows how to read CSV-files. My main responsibility
42   * is to perform file-system read operation and transform the file lines into a format suitable for
43   * constructing {@link DecisionTable}s.
44   *
45   * @since 0.2
46   */
47  public final class PlainTextContentReader implements ContentReader {
48      /**
49       * Path to a folder with source files.
50       */
51      private final URI folder;
52  
53      /**
54       * The extension of the source files.
55       */
56      private final String extension;
57  
58      /**
59       * The symbol that separates CSV-record fields.
60       */
61      private final String delimiter;
62  
63      /**
64       * Ctor with specified delimiter.
65       *
66       * @param dir The folder of source data files.
67       * @param extension The extension of source data files.
68       * @param delimiter The symbol that separates CSV-record fields.
69       */
70      public PlainTextContentReader(final URI dir, final String extension, final String delimiter) {
71          this.folder = dir;
72          this.extension = extension;
73          this.delimiter = delimiter;
74      }
75  
76      @Override
77      public List<RawContent> readAllTables() {
78          List<RawContent> contents;
79          try (Stream<Path> files = Files.walk(Paths.get(this.folder))) {
80              contents = this.contentsOf(files);
81          } catch (final IOException exception) {
82              contents = Collections.emptyList();
83          }
84          return contents;
85      }
86  
87      @Override
88      public Map<String, Locator> allTables() {
89          return this.readAllTables()
90              .stream()
91              .collect(Collectors.toMap(RawContent::tableName, RawContent::asDecisionTable));
92      }
93  
94      /**
95       * Reads the contents of all the files, passed in as a stream of paths.
96       *
97       * @param files Stream of paths to read data from.
98       * @return A collection of {@link RawContent}, each element corresponds to a single
99       *  {@link DecisionTable}.
100      */
101     private List<RawContent> contentsOf(final Stream<Path> files) {
102         return files.filter(Files::isRegularFile)
103             .filter(path -> path.getFileName().toString().endsWith(this.extension))
104             .map(Path::toString)
105             .map(this::readContentFromFile)
106             .collect(Collectors.toList());
107     }
108 
109     /**
110      * Reads the contents of the specified file as a {@link RawContent}.
111      *
112      * @param file File to read, represented as {@code String}.
113      * @return The file's contents in a format suitable for constructing {@link DecisionTable}s.
114      */
115     private RawContent readContentFromFile(final String file) {
116         final Path path = Path.of(file);
117         final List<String> contents = contentsOf(path);
118         return new RawContent(
119             this.conditionsFrom(contents),
120             this.outcomesFrom(contents),
121             path.getFileName().toString().replace(this.extension, "")
122         );
123     }
124 
125     /**
126      * Creates the 2D-array representation of the {@link DecisionTable}'s Conditions part.
127      *
128      * @param contents Raw contents of a file as a Strings collection.
129      * @return The 2D-array of Strings that holds the {@link DecisionTable}'s Conditions.
130      */
131     private String[][] outcomesFrom(final List<String> contents) {
132         return this.toArray(contents.subList(1 + separatorIndex(contents), contents.size()));
133     }
134 
135     /**
136      * Creates the 2D-array representation of the {@link DecisionTable}'s Outcomes part.
137      *
138      * @param contents Raw contents of a file as a Strings collection.
139      * @return The 2D-array of Strings that holds the {@link DecisionTable}'s Outcomes.
140      */
141     private String[][] conditionsFrom(final List<String> contents) {
142         return this.toArray(contents.subList(0, separatorIndex(contents)));
143     }
144 
145     /**
146      * Finds the position of the line separating Conditions from Outcomes in a source file.
147      *
148      * @param contents A collection of source data file lines.
149      * @return An {@code Integer} corresponding to the separating line index.
150      */
151     private static int separatorIndex(final List<String> contents) {
152         return contents.indexOf("---");
153     }
154 
155     /**
156      * Transforms the list of Strings to a 2D-array of Strings suitable for {@link DecisionTable}
157      * generation.
158      *
159      * @param lines A collection of Strings representing the file contents.
160      * @return A 2D-array of Strings
161      */
162     private String[][] toArray(final List<String> lines) {
163         final int width = lines.get(0).split(this.delimiter).length;
164         final String[][] result = new String[lines.size()][width];
165         for (int idx = 0; idx < lines.size(); idx = idx + 1) {
166             result[idx] = lines.get(idx).split(this.delimiter);
167         }
168         return result;
169     }
170 
171     /**
172      * Reads the contents of the specified file as a collection of {@code String}s.
173      *
174      * @param file File to read, represented as {@code Path}.
175      * @return The collection of {@code String}s read from file.
176      */
177     private static List<String> contentsOf(final Path file) {
178         List<String> strings;
179         try {
180             strings = Files.readAllLines(file, StandardCharsets.UTF_8);
181         } catch (final IOException exception) {
182             strings = Collections.emptyList();
183         }
184         return strings;
185     }
186 }
